 54def load_column_mapping(mapping_file_path: Path) -> Dict[str, Dict[str, str]]:
 55    """
 56    Loads the CSV to SQL column mapping from the JSON configuration file.
 57
 58    Args:
 59        mapping_file_path: Path to the column mapping JSON file.
 60
 61    Returns:
 62        A dictionary representing the column mapping.
 63
 64    Raises:
 65        FileNotFoundError: If the mapping file is not found.
 66        json.JSONDecodeError: If the mapping file is not valid JSON.
 67        SQLExportError: For other loading issues.
 68    """
 69    logger.info(f"{LOG_PREFIX}: Loading column mapping from {mapping_file_path}")
 70    if not mapping_file_path.is_file():
 71        logger.error(
 72            f"{LOG_PREFIX}: Column mapping file not found: {mapping_file_path}"
 73        )
 74        raise FileNotFoundError(f"Column mapping file not found: {mapping_file_path}")
 75    try:
 76        with open(mapping_file_path, "r", encoding="utf-8") as f:
 77            mapping = json.load(f)
 78        logger.info(f"{LOG_PREFIX}: Column mapping loaded successfully.")
 79        return mapping
 80    except json.JSONDecodeError as e:
 81        logger.error(
 82            f"{LOG_PREFIX}: Invalid JSON in column mapping file {mapping_file_path}: {e}"
 83        )
 84        raise
 85    except Exception as e:
 86        logger.error(
 87            f"{LOG_PREFIX}: Error loading column mapping file {mapping_file_path}: {e}"
 88        )
 89        raise SQLExportError(f"Could not load column mapping: {e}")

 92def get_current_offset(offset_file: Path) -> int:
 93    """
 94    Reads the current offset from the state file.
 95
 96    Args:
 97        offset_file: Path to the offset file.
 98
 99    Returns:
100        The current offset (file position), or 0 if the file doesn't exist or is invalid.
101    """
102    logger.debug(f"{LOG_PREFIX}: Reading offset from {offset_file}")
103    if not offset_file.is_file():
104        logger.info(
105            f"{LOG_PREFIX}: Offset file {offset_file} not found, starting from beginning (offset 0)."
106        )
107        return 0
108    try:
109        with open(offset_file, "r", encoding="utf-8") as f:
110            offset_str = f.read().strip()
111            offset = int(offset_str)
112            logger.info(
113                f"{LOG_PREFIX}: Successfully read offset: {offset} from {offset_file}."
114            )
115            return offset
116    except ValueError:
117        logger.warning(
118            f"{LOG_PREFIX}: Invalid content in offset file {offset_file}. Resetting to 0.",
119            exc_info=True,
120        )
121        return 0
122    except Exception as e:
123        logger.error(
124            f"{LOG_PREFIX}: Error reading offset file {offset_file}: {e}. Resetting to 0.",
125            exc_info=True,
126        )
127        return 0

155def validate_csv_header(
156    header: List[str], column_mapping: Dict[str, Dict[str, str]], csv_path: Path
157) -> None:
158    """
159    Validates the CSV header against the expected CSV columns in the mapping.
160    Note: This is a basic check. The actual CSV file might have more columns
161    than what is defined in the SQL mapping. This function checks if all
162    CSV columns *needed* for the SQL export are present.
163
164    Args:
165        header: The list of column names from the CSV file header.
166        column_mapping: The loaded column mapping.
167        csv_path: Path to the CSV file (for logging).
168
169    Raises:
170        CSVSchemaError: If any required CSV column is missing.
171    """
172    logger.debug(f"{LOG_PREFIX}: Validating CSV header for {csv_path}")
173    expected_csv_columns = set()
174    for sql_col_info in column_mapping.values():
175        csv_col_name = sql_col_info.get("csv_column_name")
176        if csv_col_name and csv_col_name != "csv_id_placeholder":  # id is special
177            expected_csv_columns.add(csv_col_name)
178
179    missing_columns = expected_csv_columns - set(header)
180    if missing_columns:
181        msg = f"CSV file {csv_path} is missing required columns: {missing_columns}. Found header: {header}"
182        logger.error(f"{LOG_PREFIX}: {msg}")
183        raise CSVSchemaError(msg)
184    logger.info(f"{LOG_PREFIX}: CSV header validation passed for {csv_path}.")

191    Args:
192        value: The string to escape.
193
194    Returns:
195        The escaped string.

202def format_sql_value(value: Any, sql_type_def: str) -> str:
203    """
204    Formats a Python value into an SQL-compatible string based on its type.
205    Handles None as NULL. For ENUMs, it assumes the value is already a string
206    that will be quoted.
207
208    Args:
209        value: The value to format.
210        sql_type_def: The SQL column definition string (e.g., "VARCHAR(50) NOT NULL", "INT UNSIGNED").
211
212    Returns:
213        SQL-formatted string representation of the value.
214    """
215    if (
216        value is None
217        or str(value).strip().lower() == "null"
218        or str(value).strip() == ""
219    ):
220        # Allow 'DEFAULT NULL' columns to receive NULL
221        if (
222            "DEFAULT NULL" in sql_type_def.upper()
223            or "PRIMARY KEY" not in sql_type_def.upper()
224        ):  # Quick check
225            return "NULL"
226        # If it's a NOT NULL column without default and value is empty/None, this is an issue
227        # The calling code should ideally handle this by providing a default or raising error
228        # For now, if it's a string type, represent as empty string, otherwise problem.
229        if "CHAR" in sql_type_def.upper() or "TEXT" in sql_type_def.upper():
230            return "''"
231        # This will likely cause an SQL error if the column is NOT NULL
232        # Consider raising an error here if value is None and column is NOT NULL without default
233        logger.warning(
234            f"{LOG_PREFIX}: Null/empty value encountered for a potentially NOT NULL column: {sql_type_def} (value: {value})"
235        )
236        return "NULL"  # Or raise error
237
238    sql_type_lower = sql_type_def.lower()
239
240    if "int" in sql_type_lower or "serial" in sql_type_lower:
241        try:
242            return str(int(value))
243        except ValueError:
244            logger.warning(
245                f"{LOG_PREFIX}: Could not convert '{value}' to int for SQL; using NULL. Column: {sql_type_def}"
246            )
247            return "NULL"  # Or raise error
248    elif "datetime" in sql_type_lower or "timestamp" in sql_type_lower:
249        # Assuming value is already in 'YYYY-MM-DD HH:MM:SS' format or a datetime object
250        # For SQLite, it's typically a string.
251        if isinstance(value, datetime.datetime):
252            return f"'{value.strftime('%Y-%m-%d %H:%M:%S')}'"
253        return escape_sql_string(str(value))  # Assuming pre-formatted string
254    elif (
255        "char" in sql_type_lower or "text" in sql_type_lower or "enum" in sql_type_lower
256    ):
257        return escape_sql_string(str(value))
258    elif "bool" in sql_type_lower:  # SQLite stores booleans as integers 0 or 1
259        return "1" if str(value).lower() in ["true", "1", "yes", "on"] else "0"
260    else:  # Default to string escaping for unknown types (e.g. IP, custom types)
261        return escape_sql_string(str(value))

264def generate_insert_statement(
265    row_dict: Dict[str, Any], table_name: str, column_mapping: Dict[str, Dict[str, str]]
266) -> Optional[str]:
267    """
268    Generates an SQL INSERT statement from a CSV row dictionary.
269
270    Args:
271        row_dict: A dictionary representing a row from the CSV (header: value).
272        table_name: The name of the SQL table to insert into.
273        column_mapping: The column mapping dictionary.
274
275    Returns:
276        A string containing the SQL INSERT statement, or None if a row is skipped.
277    """
278    # Determine target SQL columns and their corresponding values from the row_dict
279    sql_columns = []
280    sql_values = []
281
282    valid_row = True
283    for sql_col_name, mapping_info in column_mapping.items():
284        csv_col_name = mapping_info.get("csv_column_name")
285        sql_col_def = mapping_info.get("sql_column_def", "")
286
287        if sql_col_name == "id" and "AUTO_INCREMENT" in sql_col_def.upper():
288            # Skip ID column if it's auto-incrementing; DB will handle it.
289            # Alternatively, if CSV provides an ID, it should be used, and AUTO_INCREMENT removed from SQL def.
290            # For now, assuming DB generates ID.
291            continue
292
293        if not csv_col_name:
294            logger.warning(
295                f"{LOG_PREFIX}: No CSV column specified for SQL column '{sql_col_name}'. Skipping this column."
296            )
297            continue
298
299        raw_value = row_dict.get(csv_col_name)
300
301        # Basic validation: if column is NOT NULL and has no DEFAULT, raw_value must exist
302        # This is a simplified check; actual NOT NULL check depends on precise SQL definition
303        if (
304            raw_value is None
305            and "NOT NULL" in sql_col_def.upper()
306            and "DEFAULT" not in sql_col_def.upper()
307            and "AUTO_INCREMENT" not in sql_col_def.upper()
308        ):
309            logger.error(
310                f"{LOG_PREFIX}: Missing value for NOT NULL column '{sql_col_name}' (mapped from CSV '{csv_col_name}'). Row: {row_dict}. Skipping row."
311            )
312            valid_row = False
313            break  # Skip this row
314
315        formatted_value = format_sql_value(raw_value, sql_col_def)
316
317        sql_columns.append(sql_col_name)
318        sql_values.append(formatted_value)
319
320    if not valid_row or not sql_columns:
321        return None
322
323    columns_str = ", ".join(sql_columns)
324    values_str = ", ".join(sql_values)
325
326    return f"INSERT INTO {table_name} ({columns_str}) VALUES ({values_str});"

329def run_sql_export(config: AppConfig, output_log_level: str = "INFO") -> bool:
330    """
331    Main function to perform the SQL export process.
332
333    Args:
334        config: The application configuration object (AppConfig).
335        output_log_level: The desired log level for console output during this process.
336                         Note: Actual logging level is typically set globally by setup_logging.
337
338    Returns:
339        True if export completed successfully, False otherwise.
340    """
341    logger.info(f"{LOG_PREFIX}: Starting SQL export process.")
342
343    # Use attributes directly from AppConfig instance
344    csv_file_path = config.working_dir / config.csv_filename
345    state_dir_path = config.state_dir
346    offset_file_path = state_dir_path / OFFSET_FILENAME
347
348    raw_mapping_path_str = config.sql_column_mapping_file_path_str
349    column_mapping = None
350    loaded_mapping_source = ""  # For logging
351
352    if raw_mapping_path_str and raw_mapping_path_str.strip():
353        # User has specified a path, attempt to load it.
354        logger.info(
355            f"{LOG_PREFIX}: User-specified column mapping file path: '{raw_mapping_path_str}'"
356        )
357        user_mapping_file_p = Path(raw_mapping_path_str)
358        if not user_mapping_file_p.is_absolute():
359            if config.config_path and config.config_path.is_file():
360                user_mapping_file_p = config.config_path.parent / raw_mapping_path_str
361                logger.debug(
362                    f"{LOG_PREFIX}: Resolved relative user mapping path to: {user_mapping_file_p} (relative to config file {config.config_path})"
363                )
364            else:
365                user_mapping_file_p = Path.cwd() / raw_mapping_path_str
366                logger.debug(
367                    f"{LOG_PREFIX}: Resolved relative user mapping path to: {user_mapping_file_p} (relative to CWD)"
368                )
369
370        final_user_path = user_mapping_file_p.resolve()
371        logger.info(
372            f"{LOG_PREFIX}: Attempting to load user-specified mapping file from: {final_user_path}"
373        )
374        try:
375            column_mapping = load_column_mapping(final_user_path)
376            loaded_mapping_source = f"user-specified file: {final_user_path}"
377        except FileNotFoundError:
378            logger.critical(
379                f"{LOG_PREFIX}: User-specified column mapping file NOT FOUND: {final_user_path}. Aborting SQL export."
380            )
381            return False
382        except json.JSONDecodeError as e:
383            logger.critical(
384                f"{LOG_PREFIX}: Invalid JSON in user-specified column mapping file {final_user_path}: {e}. Aborting SQL export."
385            )
386            return False
387        except Exception as e:
388            logger.critical(
389                f"{LOG_PREFIX}: Error loading user-specified column mapping file {final_user_path}: {e}. Aborting SQL export.",
390                exc_info=True,
391            )
392            return False
393    else:
394        # No user path specified, use bundled default.
395        logger.info(
396            f"{LOG_PREFIX}: No user-specific column mapping file configured, attempting to load bundled default."
397        )
398        try:
399            with importlib.resources.files("lib.maillogsentinel.data").joinpath(
400                "maillogsentinel_sql_column_mapping.json"
401            ) as bundled_path:
402                if (
403                    not bundled_path.is_file()
404                ):  # Should not happen if packaged correctly
405                    logger.critical(
406                        f"{LOG_PREFIX}: Bundled column mapping file not found at expected location via importlib.resources. Path: {bundled_path}. This indicates a packaging issue. Aborting."
407                    )
408                    return False
409                column_mapping = load_column_mapping(
410                    Path(bundled_path)
411                )  # load_column_mapping expects a Path object
412                loaded_mapping_source = f"bundled default: lib.maillogsentinel.data/maillogsentinel_sql_column_mapping.json (resolved to {bundled_path})"
413        except ModuleNotFoundError:  # If lib.maillogsentinel.data is not a package
414            logger.critical(
415                f"{LOG_PREFIX}: Could not find the 'lib.maillogsentinel.data' package for bundled resources. This indicates a packaging issue. Aborting.",
416                exc_info=True,
417            )
418            return False
419        except (
420            FileNotFoundError
421        ):  # Should be caught by the bundled_path.is_file() check ideally
422            logger.critical(
423                f"{LOG_PREFIX}: Bundled column mapping file not found via importlib.resources. This indicates a packaging issue. Aborting.",
424                exc_info=True,
425            )
426            return False
427        except Exception as e:
428            logger.critical(
429                f"{LOG_PREFIX}: Error loading bundled default column mapping file: {e}. Aborting SQL export.",
430                exc_info=True,
431            )
432            return False
433
434    if (
435        column_mapping is None
436    ):  # Should have been caught by earlier returns, but as a safeguard
437        logger.critical(
438            f"{LOG_PREFIX}: Column mapping could not be loaded (reason unspecified, this is an unexpected state). Aborting."
439        )
440        return False
441
442    logger.info(
443        f"{LOG_PREFIX}: Successfully loaded column mapping from {loaded_mapping_source}."
444    )
445
446    sql_output_dir = config.working_dir / SQL_EXPORT_SUBDIR
447    table_name = config.sql_target_table_name
448
449    logger.info(f"{LOG_PREFIX}: CSV source: {csv_file_path}")
450    logger.info(f"{LOG_PREFIX}: State directory: {state_dir_path}")
451    logger.info(f"{LOG_PREFIX}: Offset file: {offset_file_path}")
452    # No longer log final_mapping_file_path here, use loaded_mapping_source instead
453    logger.info(f"{LOG_PREFIX}: SQL output directory: {sql_output_dir}")
454    logger.info(f"{LOG_PREFIX}: Target SQL table name: {table_name}")
455
456    if not csv_file_path.is_file():
457        logger.error(
458            f"{LOG_PREFIX}: CSV file {csv_file_path} not found. Aborting SQL export."
459        )
460        return False
461
462    try:
463        sql_output_dir.mkdir(parents=True, exist_ok=True)
464    except OSError as e:
465        logger.critical(
466            f"{LOG_PREFIX}: Failed to create SQL output directory {sql_output_dir}: {e}. Aborting.",
467            exc_info=True,
468        )
469        return False
470
471    current_offset = get_current_offset(offset_file_path)
472    new_offset = current_offset
473    records_processed = 0
474    records_exported = 0
475
476    # Prepare output SQL file
477    timestamp_str = datetime.datetime.now().strftime("%Y%m%d_%H%M")
478    # Original request: YYYYMMDD_HH:MM_maillogsentinel_export.sql. Colon is problematic in filenames.
479    # Using YYYYMMDD_HHMM_maillogsentinel_export.sql
480    timestamp_fn_str = datetime.datetime.now().strftime("%Y%m%d_%H%M")
481    sql_file_name = f"{timestamp_fn_str}_maillogsentinel_export.sql"
482    sql_file_path = sql_output_dir / sql_file_name
483
484    logger.info(
485        f"{LOG_PREFIX}: Exporting new records from {csv_file_path} to {sql_file_path}"
486    )
487
488    try:
489        with open(csv_file_path, "r", encoding="utf-8", newline="") as infile, open(
490            sql_file_path, "w", encoding="utf-8"
491        ) as outfile:
492
493            infile.seek(current_offset)
494
495            # Check if we are past the header (if offset is not 0)
496            # If offset is 0, we read the header. If > 0, we assume header was processed.
497            header = None
498            if current_offset == 0:
499                first_line = infile.readline()
500                if not first_line:
501                    logger.info(
502                        f"{LOG_PREFIX}: CSV file {csv_file_path} is empty. No data to export."
503                    )
504                    # update_offset(offset_file_path, new_offset) # Offset remains 0
505                    # It's better to remove the empty SQL file
506                    outfile.close()  # Close before attempting to delete
507                    sql_file_path.unlink(missing_ok=True)
508                    return True
509
510                new_offset += len(
511                    first_line.encode("utf-8")
512                )  # Update offset past header
513                header = [h.strip() for h in first_line.strip().split(";")]
514                try:
515                    validate_csv_header(header, column_mapping, csv_file_path)
516                except CSVSchemaError as e:
517                    logger.error(
518                        f"{LOG_PREFIX}: CSV header validation failed: {e}. Aborting."
519                    )
520                    # Consider what to do with sql_file_path - it might be empty.
521                    # Clean up potentially empty or partially written SQL file
522                    outfile.close()
523                    sql_file_path.unlink(missing_ok=True)
524                    return False
525            else:
526                # We don't have the header if resuming. This could be an issue if
527                # validate_csv_header is crucial for every run or if column order changes.
528                # For now, assume structure is stable. A robust way is to always read header
529                # or store expected header hash.
530                # For simplicity, we'll rely on column_mapping's csv_column_name for dict keys.
531                # This means we need the header for dictreader.
532                # Solution: If resuming, we still need the header. Store it or re-read it.
533                # Let's re-read the header line without advancing the main processing offset.
534                current_pos = infile.tell()
535                infile.seek(0)
536                header_line_for_resume = infile.readline()
537                if not header_line_for_resume:
538                    logger.error(
539                        f"{LOG_PREFIX}: CSV file {csv_file_path} seems to contain no header. Aborting."
540                    )
541                    outfile.close()
542                    sql_file_path.unlink(missing_ok=True)
543                    return False
544                header = [h.strip() for h in header_line_for_resume.strip().split(";")]
545                infile.seek(current_pos)  # Return to where we were
546                try:
547                    validate_csv_header(header, column_mapping, csv_file_path)
548                except CSVSchemaError as e:
549                    logger.error(
550                        f"{LOG_PREFIX}: CSV header validation failed on resume: {e}. Aborting."
551                    )
552                    outfile.close()
553                    sql_file_path.unlink(missing_ok=True)
554                    return False
555
556            reader = csv.DictReader(infile, delimiter=";", fieldnames=header)
557
558            outfile.write("BEGIN TRANSACTION;\n")
559
560            for row in reader:
561                records_processed += 1
562                # Check for empty rows (e.g. just delimiters ;;;;)
563                if not any(row.values()):
564                    logger.debug(
565                        f"{LOG_PREFIX}: Skipping empty or malformed row: {row}"
566                    )
567                    # Still need to update offset for this line
568                    # The DictReader handles line ending consumption.
569                    # To get the byte length of the line for offset:
570                    # This is tricky with DictReader. Simplest is to read line by line first,
571                    # then parse. For now, this part of offset update is imprecise with DictReader.
572                    # A more robust offset: re-open and read line-by-line to count bytes.
573                    # For now, new_offset will be updated at the end based on infile.tell().
574                    continue
575
576                try:
577                    # Ensure all expected keys are present in row, map to None if not
578                    # This is important if CSV is sparse or has missing optional fields
579                    # Handled by row_dict.get(csv_col_name) in generate_insert_statement
580                    insert_stmt = generate_insert_statement(
581                        row, table_name, column_mapping
582                    )
583                    if insert_stmt:
584                        outfile.write(insert_stmt + "\n")
585                        records_exported += 1
586                except Exception as e:
587                    logger.error(
588                        f"{LOG_PREFIX}: Error processing row: {row}. Error: {e}",
589                        exc_info=True,
590                    )
591                    # Decide if we skip this row or abort. For now, skip.
592
593            # After processing all available lines from the current offset
594            new_offset = infile.tell()  # Get the end position
595
596            outfile.write("COMMIT;\n")
597            logger.info(
598                f"{LOG_PREFIX}: SQL export process finished. Processed: {records_processed} lines. Exported: {records_exported} records."
599            )
600

602        logger.critical(
603            f"{LOG_PREFIX}: CSV file {csv_file_path} not found during processing. Aborting.",
604            exc_info=True,
605        )
606        if "outfile" in locals() and not outfile.closed:
607            outfile.close()
608        if sql_file_path.exists():
609            sql_file_path.unlink(missing_ok=True)  # cleanup
610        return False
611    except CSVSchemaError as e:  # Already logged by validate_csv_header
612        # Cleanup already handled in validate_csv_header's calling block
613        return False
614    except Exception as e:
615        logger.critical(
616            f"{LOG_PREFIX}: Unexpected error during SQL export: {e}", exc_info=True
617        )
618        if "outfile" in locals() and not outfile.closed:
619            outfile.close()
620        if sql_file_path.exists():
621            sql_file_path.unlink(missing_ok=True)  # cleanup
622        return False
623    finally:
624        if "infile" in locals() and not infile.closed:
625            # Ensure new_offset is captured if loop was exited prematurely by error after some reads
626            new_offset = infile.tell()
627            infile.close()
628        if "outfile" in locals() and not outfile.closed:
629            outfile.close()
630
631    if records_exported == 0:
632        if records_processed > 0:
633            # Processed lines but exported nothing (e.g., all rows skipped due to errors/filters)
634            logger.warning(
635                f"{LOG_PREFIX}: Processed {records_processed} lines, but no records were actually exported. SQL file {sql_file_path} will contain only BEGIN/COMMIT. This might indicate data or mapping issues."
636            )
637            # Keep the file for inspection in this specific case.
638        else:
639            # No records exported AND no records processed (beyond header if it was the first run)
640            # This includes:
641            # 1. Truly empty CSV (already handled by returning after unlink if first_line is empty)
642            # 2. Header-only CSV on first run (current_offset=0 initially, records_processed=0)
643            # 3. Resume run with no new data lines (current_offset>0 initially, records_processed=0)
644            logger.info(
645                f"{LOG_PREFIX}: No records exported and no new data lines processed. Removing SQL export file: {sql_file_path}"
646            )
647            sql_file_path.unlink(missing_ok=True)
648    else:  # records_exported > 0
649        logger.info(
650            f"{LOG_PREFIX}: Successfully created SQL export file: {sql_file_path} with {records_exported} records."
651        )
652
653    # Update offset only if processing was generally successful or partially successful
654    # If a critical error happened early (e.g. cant load mapping), offset should not change.
655    # Current logic updates offset if we reach here.
656    # If CSV was not found, we return False before this.
657    # If mapping failed, we return False before this.
658    update_offset(offset_file_path, new_offset)
659
660    logger.info(
661        f"{LOG_PREFIX}: SQL export process complete. Final offset: {new_offset}"
662    )
