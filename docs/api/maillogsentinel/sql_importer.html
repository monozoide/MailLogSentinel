maillogsentinel.sql_importer

Handles the SQL import functionality for MailLogSentinel.

Scans for .sql files generated by the export process, imports them into the SQLite3 database, manages concurrency with a lock file, and handles errors with retries and rollbacks.

 52class FileLock:
 53    """
 54    A simple file-based lock for POSIX systems using fcntl.
 55    Ensures only one instance of the import process runs.
 56    """
 57
 58    def __init__(self, lock_file_path: Path):
 59        self.lock_file_path = lock_file_path
 60        self._lock_file_handle = None
 61
 62    def acquire(self) -> bool:
 63        """
 64        Acquires the lock. Returns True if successful, False otherwise.
 65        """
 66        logger.debug(f"{LOG_PREFIX}: Attempting to acquire lock: {self.lock_file_path}")
 67        try:
 68            self.lock_file_path.parent.mkdir(parents=True, exist_ok=True)
 69            # Open in append mode so it doesn't truncate if it exists; content doesn't matter.
 70            self._lock_file_handle = open(self.lock_file_path, "a")
 71            fcntl.flock(self._lock_file_handle.fileno(), fcntl.LOCK_EX | fcntl.LOCK_NB)
 72            logger.info(f"{LOG_PREFIX}: Lock acquired: {self.lock_file_path}")
 73            # Write PID to lock file for informational purposes
 74            self._lock_file_handle.seek(0)
 75            self._lock_file_handle.truncate()
 76            self._lock_file_handle.write(
 77                str(Path.cwd().owner()) + "\n"
 78            )  # For debugging who owns it
 79            self._lock_file_handle.write(str(os.getpid()) + "\n")
 80            self._lock_file_handle.flush()
 81            return True
 82        except (IOError, BlockingIOError):
 83            logger.warning(
 84                f"{LOG_PREFIX}: Lock {self.lock_file_path} is already held by another process."
 85            )
 86            if self._lock_file_handle:
 87                self._lock_file_handle.close()
 88                self._lock_file_handle = None
 89            return False
 90        except Exception as e:
 91            logger.error(
 92                f"{LOG_PREFIX}: Error acquiring lock {self.lock_file_path}: {e}",
 93                exc_info=True,
 94            )
 95            if self._lock_file_handle:
 96                self._lock_file_handle.close()
 97                self._lock_file_handle = None
 98            # Re-raise as a LockError for clarity in the calling function
 99            raise LockError(f"Failed to acquire lock: {e}")
100
101    def release(self) -> None:
102        """
103        Releases the lock.
104        """
105        if self._lock_file_handle:
106            logger.debug(f"{LOG_PREFIX}: Releasing lock: {self.lock_file_path}")
107            fcntl.flock(self._lock_file_handle.fileno(), fcntl.LOCK_UN)
108            self._lock_file_handle.close()
109            self._lock_file_handle = None
110            # Optionally remove the lock file, though not strictly necessary
111            # self.lock_file_path.unlink(missing_ok=True)
112            logger.info(f"{LOG_PREFIX}: Lock released: {self.lock_file_path}")
113
114    def __enter__(self):
115        if not self.acquire():
116            raise LockError(f"Could not acquire lock: {self.lock_file_path}")
117        return self
118
119    def __exit__(self, exc_type, exc_val, exc_tb):
120        self.release()

123def get_db_connection(db_path: Path) -> sqlite3.Connection:
124    """
125    Establishes a connection to the SQLite database.
126
127    Args:
128        db_path: Path to the SQLite database file.
129
130    Returns:
131        An sqlite3.Connection object.
132
133    Raises:
134        DatabaseError: If the connection cannot be established.
135    """
136    logger.debug(f"{LOG_PREFIX_DB}: Connecting to database: {db_path}")
137    try:
138        db_path.parent.mkdir(parents=True, exist_ok=True)
139        conn = sqlite3.connect(str(db_path), timeout=10)  # Increased timeout
140        logger.info(f"{LOG_PREFIX_DB}: Successfully connected to database: {db_path}")
141        return conn
142    except sqlite3.Error as e:
143        logger.error(
144            f"{LOG_PREFIX_DB}: Failed to connect to database {db_path}: {e}",
145            exc_info=True,
146        )
147        raise DatabaseError(f"Could not connect to SQLite database: {e}")

150def create_table_if_not_exists(
151    conn: sqlite3.Connection, table_name: str, column_mapping_file: Path
152) -> None:
153    """
154    Creates the target table in the database if it doesn't already exist,
155    based on the SQL column definitions in the mapping file.
156
157    Args:
158        conn: Active SQLite database connection.
159        table_name: Name of the table to create.
160        column_mapping_file: Path to the JSON column mapping file.
161
162    Raises:
163        DatabaseError: If table creation fails.
164    """
165    logger.info(f"{LOG_PREFIX_DB}: Ensuring table '{table_name}' exists...")
166    try:
167        # load_column_mapping is from sql_exporter, its internal logs will use its own prefix.
168        mapping = load_column_mapping(column_mapping_file)
169
170        # Construct CREATE TABLE statement
171        # Example: "id INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY" becomes "id INTEGER PRIMARY KEY AUTOINCREMENT" for SQLite
172        # SQLite types are a bit different: TEXT, INTEGER, REAL, BLOB, NULL
173        # AUTO_INCREMENT is handled by INTEGER PRIMARY KEY AUTOINCREMENT
174        # VARCHAR(N) becomes TEXT
175        # DATETIME becomes TEXT (ISO8601 strings recommended)
176        # ENUM becomes TEXT with a CHECK constraint
177
178        column_defs_sql = []
179        for sql_col_name, info in mapping.items():
180            sql_def = info.get("sql_column_def", "").upper()
181            sqlite_col_def = sql_col_name  # Start with name
182
183            if (
184                "PRIMARY KEY AUTO_INCREMENT" in sql_def
185                or "AUTO_INCREMENT PRIMARY KEY" in sql_def
186            ):
187                # SQLite specific for auto-incrementing primary key
188                sqlite_col_def += " INTEGER PRIMARY KEY AUTOINCREMENT"
189            else:
190                if "INT" in sql_def:
191                    sqlite_col_def += " INTEGER"
192                elif "CHAR" in sql_def:
193                    sqlite_col_def += " TEXT"  # VARCHAR, CHAR -> TEXT
194                elif "TEXT" in sql_def:
195                    sqlite_col_def += " TEXT"
196                elif "DATETIME" in sql_def:
197                    sqlite_col_def += " TEXT"  # Store as ISO8601 strings
198                elif "ENUM" in sql_def:
199                    sqlite_col_def += " TEXT"
200                    # Extract enum values for CHECK constraint, e.g., ENUM('OK', 'Errno 1')
201                    try:
202                        enum_values_str = sql_def.split("ENUM(")[1].split(")")[0]
203                        enum_values = [
204                            val.strip().strip("'") for val in enum_values_str.split(",")
205                        ]
206                        check_values = ", ".join([f"'{v}'" for v in enum_values])
207                        sqlite_col_def += f" CHECK ({sql_col_name} IN ({check_values}))"
208                    except IndexError:
209                        logger.warning(
210                            f"{LOG_PREFIX_DB}: Could not parse ENUM values for {sql_col_name} from '{sql_def}'. No CHECK constraint added."
211                        )
212                else:  # Default to TEXT for unknown types
213                    sqlite_col_def += " TEXT"
214
215                if "NOT NULL" in sql_def:
216                    sqlite_col_def += " NOT NULL"
217                if "DEFAULT NULL" in sql_def:
218                    pass  # SQLite columns are nullable by default unless NOT NULL is specified
219                elif "DEFAULT " in sql_def:  # e.g. DEFAULT 'some_value' or DEFAULT 0
220                    default_val_str = sql_def.split("DEFAULT ")[1].split(" ")[
221                        0
222                    ]  # Basic parse
223                    sqlite_col_def += f" DEFAULT {default_val_str}"  # Assumes default_val_str is correctly quoted if string
224
225            column_defs_sql.append(sqlite_col_def)
226
227        create_table_sql = (
228            f"CREATE TABLE IF NOT EXISTS {table_name} ({', '.join(column_defs_sql)});"
229        )
230        logger.debug(
231            f"{LOG_PREFIX_DB}: Executing CREATE TABLE statement: {create_table_sql}"
232        )
233
234        cursor = conn.cursor()
235        cursor.execute(create_table_sql)
236        conn.commit()
237        logger.info(f"{LOG_PREFIX_DB}: Table '{table_name}' ensured to exist.")
238    except sqlite3.Error as e:
239        logger.error(
240            f"{LOG_PREFIX_DB}: Database error ensuring table '{table_name}' exists: {e}",
241            exc_info=True,
242        )
243        raise DatabaseError(f"Failed to create/verify table '{table_name}': {e}")
244    except Exception as e:  # Catch errors from load_column_mapping or parsing
245        logger.error(
246            f"{LOG_PREFIX_DB}: Error preparing CREATE TABLE statement for '{table_name}': {e}",
247            exc_info=True,
248        )
249        raise DatabaseError(f"Could not prepare CREATE TABLE for '{table_name}': {e}")

252def get_already_imported_files(imported_log_path: Path) -> set[str]:
253    """Reads the list of already imported .sql file basenames."""
254    if not imported_log_path.is_file():
255        return set()
256    try:
257        with open(imported_log_path, "r", encoding="utf-8") as f:
258            return {line.strip() for line in f if line.strip()}
259    except IOError as e:
260        logger.warning(
261            f"{LOG_PREFIX}: Could not read imported files log {imported_log_path}: {e}. Assuming no files imported via this log.",
262            exc_info=True,
263        )
264        return set()

281def run_sql_import(config: AppConfig, output_log_level: str = "INFO") -> bool:
282    """
283    Main function to perform the SQL import process.
284
285    Args:
286        config: The application configuration object (AppConfig).
287        output_log_level: The desired log level for console output.
288
289    Returns:
290        True if import completed successfully (or no new files), False otherwise.
291    """
292    logger.info(f"{LOG_PREFIX}: Starting SQL import process.")
293
294    lock_file = config.state_dir / LOCK_FILENAME
295    sql_files_dir = config.working_dir / SQL_DIR_NAME
296    db_path = Path(config.sqlite_db_path)  # Ensure it's a Path object
297    imported_log_path = config.state_dir / IMPORTED_FILES_LOG
298    table_name = (
299        config.sql_target_table_name
300    )  # Moved up, as it's independent of mapping path
301
302    # --- Determine Column Mapping File Path ---
303    raw_mapping_path_str = config.sql_column_mapping_file_path_str
304    final_mapping_file_path: Optional[Path] = None
305    loaded_mapping_source_log_msg = ""
306
307    if raw_mapping_path_str and raw_mapping_path_str.strip():
308        logger.info(
309            f"{LOG_PREFIX}: User-specified column mapping file path: '{raw_mapping_path_str}'"
310        )
311        user_mapping_file_p = Path(raw_mapping_path_str)
312        if not user_mapping_file_p.is_absolute():
313            if config.config_path and config.config_path.is_file():
314                user_mapping_file_p = config.config_path.parent / raw_mapping_path_str
315                logger.debug(
316                    f"{LOG_PREFIX}: Resolved relative user mapping path to: {user_mapping_file_p} (relative to config file {config.config_path})"
317                )
318            else:
319                user_mapping_file_p = Path.cwd() / raw_mapping_path_str
320                logger.debug(
321                    f"{LOG_PREFIX}: Resolved relative user mapping path to: {user_mapping_file_p} (relative to CWD)"
322                )
323
324        final_mapping_file_path = user_mapping_file_p.resolve()
325        loaded_mapping_source_log_msg = (
326            f"user-specified file: {final_mapping_file_path}"
327        )
328        # Actual loading and FileNotFoundError for this user-specified path will be handled by create_table_if_not_exists
329        # which calls load_column_mapping. If it fails there, it's a critical error for the import.
330    else:
331        logger.info(
332            f"{LOG_PREFIX}: No user-specific column mapping file configured, attempting to load bundled default."
333        )
334        try:
335            with importlib.resources.files("lib.maillogsentinel.data").joinpath(
336                "maillogsentinel_sql_column_mapping.json"
337            ) as bundled_path_traversable:
338                final_mapping_file_path = Path(
339                    bundled_path_traversable
340                )  # Convert Traversable to Path
341                if not final_mapping_file_path.is_file():
342                    logger.critical(
343                        f"{LOG_PREFIX}: Bundled column mapping file not found at expected location via importlib.resources: {final_mapping_file_path}. This indicates a packaging issue. Aborting SQL import."
344                    )
345                    return False  # Abort early if bundled default is missing
346                loaded_mapping_source_log_msg = f"bundled default: lib.maillogsentinel.data/maillogsentinel_sql_column_mapping.json (resolved to {final_mapping_file_path})"
347        except ModuleNotFoundError:  # If lib.maillogsentinel.data is not a package
348            logger.critical(
349                f"{LOG_PREFIX}: Could not find the 'lib.maillogsentinel.data' package for bundled resources. This indicates a packaging issue. Aborting SQL import.",
350                exc_info=True,
351            )
352            return False
353        except (
354            FileNotFoundError
355        ):  # Should be caught by is_file() check, but as safeguard
356            logger.critical(
357                f"{LOG_PREFIX}: Bundled column mapping file not found via importlib.resources (FileNotFoundError). This indicates a packaging issue. Aborting SQL import.",
358                exc_info=True,
359            )
360            return False
361        except Exception as e:
362            logger.critical(
363                f"{LOG_PREFIX}: Error loading bundled default column mapping file: {e}. Aborting SQL import.",
364                exc_info=True,
365            )
366            return False
367
368    if (
369        final_mapping_file_path is None
370    ):  # Safeguard, should have been set or returned False
371        logger.critical(
372            f"{LOG_PREFIX}: Column mapping file path could not be determined. Aborting SQL import."
373        )
374        return False
375
376    logger.info(
377        f"{LOG_PREFIX}: Using column mapping from {loaded_mapping_source_log_msg} for table creation."
378    )
379    # --- End Determine Column Mapping File Path ---
380
381    importer_lock = FileLock(lock_file)
382    try:
383        with importer_lock:  # Acquires lock, raises LockError if fails
384            logger.info(f"{LOG_PREFIX}: Lock acquired. Proceeding with import.")
385
386            conn: Optional[sqlite3.Connection] = None
387            retries = 0
388            while retries <= MAX_RETRIES:
389                try:
390                    conn = get_db_connection(db_path)
391                    # Ensure table exists
392                    create_table_if_not_exists(
393                        conn, table_name, final_mapping_file_path
394                    )
395                    break  # Connection successful
396                except (
397                    DatabaseError
398                ) as e:  # Errors here will have LOG_PREFIX_DB from underlying functions
399                    logger.error(
400                        f"{LOG_PREFIX}: Database connection or initial table setup error: {e}"
401                    )  # General error for the step
402                    if retries == MAX_RETRIES:
403                        logger.critical(
404                            f"{LOG_PREFIX_DB}: Max retries reached for DB connection/setup. Aborting."
405                        )
406                        return False
407
408                    backoff_time = min(
409                        INITIAL_BACKOFF_SECONDS * (2**retries), MAX_BACKOFF_SECONDS
410                    )
411                    logger.info(
412                        f"{LOG_PREFIX_DB}: Retrying DB connection/setup in {backoff_time} seconds... (Attempt {retries + 1}/{MAX_RETRIES})"
413                    )
414                    time.sleep(backoff_time)
415                    retries += 1
416
417            if not conn:  # Should not happen if loop logic is correct
418                logger.critical(
419                    f"{LOG_PREFIX_DB}: Failed to establish database connection after retries."
420                )
421                return False
422
423            already_imported = get_already_imported_files(imported_log_path)
424
425            if not sql_files_dir.exists() or not any(sql_files_dir.iterdir()):
426                logger.info(
427                    f"{LOG_PREFIX}: SQL files directory {sql_files_dir} is empty or does not exist. No files to import."
428                )
429                if conn:
430                    conn.close()
431                return True
432
433            sql_files_to_import = sorted(
434                [
435                    f
436                    for f in sql_files_dir.glob("*.sql")
437                    if f.is_file() and f.name not in already_imported
438                ]
439            )
440
441            if not sql_files_to_import:
442                logger.info(
443                    f"{LOG_PREFIX}: No new SQL files found to import in {sql_files_dir}."
444                )
445                if conn:
446                    conn.close()
447                return True
448
449            logger.info(
450                f"{LOG_PREFIX}: Found {len(sql_files_to_import)} new SQL file(s) to import."
451            )
452            total_files_processed = 0
453            total_files_failed = 0
454
455            for sql_file in sql_files_to_import:
456                logger.info(f"{LOG_PREFIX}: Processing SQL file: {sql_file.name}")
457                try:
458                    with open(sql_file, "r", encoding="utf-8") as f_sql:
459                        sql_script = f_sql.read()
460
461                    # SQLite's executescript handles transactions within the script.
462                    # If the script has BEGIN/COMMIT, it will be atomic.
463                    # If not, executescript itself doesn't wrap in a transaction by default.
464                    # Our exporter creates files with BEGIN TRANSACTION; ... COMMIT;
465                    conn.executescript(
466                        sql_script
467                    )  # This will commit if successful, or rollback if error within script
468                    conn.commit()  # Ensure changes from executescript are committed if it doesn't do it itself for some reason (it should)
469
470                    mark_file_as_imported(imported_log_path, sql_file.name)
471                    logger.info(
472                        f"{LOG_PREFIX}: Successfully imported {sql_file.name}"
473                    )  # General import log
474                    total_files_processed += 1
475                except sqlite3.Error as e_db:
476                    logger.error(
477                        f"{LOG_PREFIX_DB}: SQLite error processing file {sql_file.name}: {e_db}. The transaction in the file should have rolled back.",
478                        exc_info=True,
479                    )
480                    # Automatic rollback on error is a feature of how SQLite handles executescript with transactional SQL files.
481                    # No explicit rollback call needed here if the .sql file itself is transactional.
482                    total_files_failed += 1
483                except (
484                    IOError
485                ) as e_io:  # This is an I/O error for reading the .sql file, not DB specific
486                    logger.error(
487                        f"{LOG_PREFIX}: IO error reading file {sql_file.name}: {e_io}",
488                        exc_info=True,
489                    )
490                    total_files_failed += 1
491                except Exception as e_gen:  # General error, could be anything
492                    logger.error(
493                        f"{LOG_PREFIX}: Unexpected error processing file {sql_file.name}: {e_gen}",
494                        exc_info=True,
495                    )
496                    total_files_failed += 1
497
498            if conn:
499                logger.debug(f"{LOG_PREFIX_DB}: Closing database connection.")
500                conn.close()
501
502            logger.info(
503                f"{LOG_PREFIX}: SQL import process finished. Processed: {total_files_processed} files. Failed: {total_files_failed} files."
504            )
505            return total_files_failed == 0
506
507    except LockError:
508        logger.warning(
509            f"{LOG_PREFIX}: Could not acquire import lock. Another import process may be running. Exiting."
510        )  # Lock related, uses general LOG_PREFIX
511        return False  # Indicate failure due to locking
512    except Exception as e:
513        logger.critical(
514            f"{LOG_PREFIX}: Unhandled exception during SQL import: {e}", exc_info=True
515        )  # General import error
516        return False
517    finally:
518        # Ensure lock is released if acquired and an unexpected error occurred before __exit__
519        if importer_lock._lock_file_handle:  # Check if lock was acquired
520            importer_lock.release()
521
522
523if __name__ == "__main__":
524    # Basic testing for sql_importer.py
525    print("Running sql_importer.py directly for testing.")
526    logging.basicConfig(
527        level=logging.DEBUG,
528        format="%(asctime)s - %(name)s - %(levelname)s - %(message)s",
529    )

 63    def acquire(self) -> bool:
 64        """
 65        Acquires the lock. Returns True if successful, False otherwise.
 66        """
 67        logger.debug(f"{LOG_PREFIX}: Attempting to acquire lock: {self.lock_file_path}")
 68        try:
 69            self.lock_file_path.parent.mkdir(parents=True, exist_ok=True)
 70            # Open in append mode so it doesn't truncate if it exists; content doesn't matter.
 71            self._lock_file_handle = open(self.lock_file_path, "a")
 72            fcntl.flock(self._lock_file_handle.fileno(), fcntl.LOCK_EX | fcntl.LOCK_NB)
 73            logger.info(f"{LOG_PREFIX}: Lock acquired: {self.lock_file_path}")
 74            # Write PID to lock file for informational purposes
 75            self._lock_file_handle.seek(0)
 76            self._lock_file_handle.truncate()
 77            self._lock_file_handle.write(
 78                str(Path.cwd().owner()) + "\n"
 79            )  # For debugging who owns it
 80            self._lock_file_handle.write(str(os.getpid()) + "\n")
 81            self._lock_file_handle.flush()
 82            return True
 83        except (IOError, BlockingIOError):
 84            logger.warning(
 85                f"{LOG_PREFIX}: Lock {self.lock_file_path} is already held by another process."
 86            )
 87            if self._lock_file_handle:
 88                self._lock_file_handle.close()
 89                self._lock_file_handle = None
 90            return False
 91        except Exception as e:
 92            logger.error(
 93                f"{LOG_PREFIX}: Error acquiring lock {self.lock_file_path}: {e}",
 94                exc_info=True,
 95            )
 96            if self._lock_file_handle:
 97                self._lock_file_handle.close()
 98                self._lock_file_handle = None
 99            # Re-raise as a LockError for clarity in the calling function
100            raise LockError(f"Failed to acquire lock: {e}")
101
102    def release(self) -> None:
103        """
104        Releases the lock.
105        """
106        if self._lock_file_handle:
107            logger.debug(f"{LOG_PREFIX}: Releasing lock: {self.lock_file_path}")
108            fcntl.flock(self._lock_file_handle.fileno(), fcntl.LOCK_UN)
109            self._lock_file_handle.close()
110            self._lock_file_handle = None
111            # Optionally remove the lock file, though not strictly necessary
112            # self.lock_file_path.unlink(missing_ok=True)
113            logger.info(f"{LOG_PREFIX}: Lock released: {self.lock_file_path}")
114
115    def __enter__(self):
116        if not self.acquire():
117            raise LockError(f"Could not acquire lock: {self.lock_file_path}")
118        return self
119
120    def __exit__(self, exc_type, exc_val, exc_tb):
121        self.release()

151def create_table_if_not_exists(
152    conn: sqlite3.Connection, table_name: str, column_mapping_file: Path
153) -> None:
154    """
155    Creates the target table in the database if it doesn't already exist,
156    based on the SQL column definitions in the mapping file.
157
158    Args:
159        conn: Active SQLite database connection.
160        table_name: Name of the table to create.
161        column_mapping_file: Path to the JSON column mapping file.
162
163    Raises:
164        DatabaseError: If table creation fails.
165    """
166    logger.info(f"{LOG_PREFIX_DB}: Ensuring table '{table_name}' exists...")
167    try:
168        # load_column_mapping is from sql_exporter, its internal logs will use its own prefix.
169        mapping = load_column_mapping(column_mapping_file)
170
171        # Construct CREATE TABLE statement
172        # Example: "id INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY" becomes "id INTEGER PRIMARY KEY AUTOINCREMENT" for SQLite
173        # SQLite types are a bit different: TEXT, INTEGER, REAL, BLOB, NULL
174        # AUTO_INCREMENT is handled by INTEGER PRIMARY KEY AUTOINCREMENT
175        # VARCHAR(N) becomes TEXT
176        # DATETIME becomes TEXT (ISO8601 strings recommended)
177        # ENUM becomes TEXT with a CHECK constraint
178
179        column_defs_sql = []
180        for sql_col_name, info in mapping.items():
181            sql_def = info.get("sql_column_def", "").upper()
182            sqlite_col_def = sql_col_name  # Start with name
183
184            if (
185                "PRIMARY KEY AUTO_INCREMENT" in sql_def
186                or "AUTO_INCREMENT PRIMARY KEY" in sql_def
187            ):
188                # SQLite specific for auto-incrementing primary key
189                sqlite_col_def += " INTEGER PRIMARY KEY AUTOINCREMENT"
190            else:
191                if "INT" in sql_def:
192                    sqlite_col_def += " INTEGER"
193                elif "CHAR" in sql_def:
194                    sqlite_col_def += " TEXT"  # VARCHAR, CHAR -> TEXT
195                elif "TEXT" in sql_def:
196                    sqlite_col_def += " TEXT"
197                elif "DATETIME" in sql_def:
198                    sqlite_col_def += " TEXT"  # Store as ISO8601 strings
199                elif "ENUM" in sql_def:
200                    sqlite_col_def += " TEXT"
201                    # Extract enum values for CHECK constraint, e.g., ENUM('OK', 'Errno 1')
202                    try:
203                        enum_values_str = sql_def.split("ENUM(")[1].split(")")[0]
204                        enum_values = [
205                            val.strip().strip("'") for val in enum_values_str.split(",")
206                        ]
207                        check_values = ", ".join([f"'{v}'" for v in enum_values])
208                        sqlite_col_def += f" CHECK ({sql_col_name} IN ({check_values}))"
209                    except IndexError:
210                        logger.warning(
211                            f"{LOG_PREFIX_DB}: Could not parse ENUM values for {sql_col_name} from '{sql_def}'. No CHECK constraint added."
212                        )
213                else:  # Default to TEXT for unknown types
214                    sqlite_col_def += " TEXT"
215
216                if "NOT NULL" in sql_def:
217                    sqlite_col_def += " NOT NULL"
218                if "DEFAULT NULL" in sql_def:
219                    pass  # SQLite columns are nullable by default unless NOT NULL is specified
220                elif "DEFAULT " in sql_def:  # e.g. DEFAULT 'some_value' or DEFAULT 0
221                    default_val_str = sql_def.split("DEFAULT ")[1].split(" ")[
222                        0
223                    ]  # Basic parse
224                    sqlite_col_def += f" DEFAULT {default_val_str}"  # Assumes default_val_str is correctly quoted if string
225
226            column_defs_sql.append(sqlite_col_def)
227
228        create_table_sql = (
229            f"CREATE TABLE IF NOT EXISTS {table_name} ({', '.join(column_defs_sql)});"
230        )
231        logger.debug(
232            f"{LOG_PREFIX_DB}: Executing CREATE TABLE statement: {create_table_sql}"
233        )
234
235        cursor = conn.cursor()
236        cursor.execute(create_table_sql)
237        conn.commit()
238        logger.info(f"{LOG_PREFIX_DB}: Table '{table_name}' ensured to exist.")
239    except sqlite3.Error as e:
240        logger.error(
241            f"{LOG_PREFIX_DB}: Database error ensuring table '{table_name}' exists: {e}",
242            exc_info=True,
243        )
244        raise DatabaseError(f"Failed to create/verify table '{table_name}': {e}")
245    except Exception as e:  # Catch errors from load_column_mapping or parsing
246        logger.error(

248            exc_info=True,
249        )

348        except ModuleNotFoundError:  # If lib.maillogsentinel.data is not a package
349            logger.critical(
350                f"{LOG_PREFIX}: Could not find the 'lib.maillogsentinel.data' package for bundled resources. This indicates a packaging issue. Aborting SQL import.",
351                exc_info=True,
352            )
353            return False
354        except (
355            FileNotFoundError
356        ):  # Should be caught by is_file() check, but as safeguard
357            logger.critical(
358                f"{LOG_PREFIX}: Bundled column mapping file not found via importlib.resources (FileNotFoundError). This indicates a packaging issue. Aborting SQL import.",
359                exc_info=True,
360            )
361            return False
362        except Exception as e:
363            logger.critical(
364                f"{LOG_PREFIX}: Error loading bundled default column mapping file: {e}. Aborting SQL import.",
365                exc_info=True,
366            )
367            return False
368
369    if (
370        final_mapping_file_path is None
371    ):  # Safeguard, should have been set or returned False
372        logger.critical(
373            f"{LOG_PREFIX}: Column mapping file path could not be determined. Aborting SQL import."
374        )
375        return False
376
377    logger.info(
378        f"{LOG_PREFIX}: Using column mapping from {loaded_mapping_source_log_msg} for table creation."
379    )
380    # --- End Determine Column Mapping File Path ---
381
382    importer_lock = FileLock(lock_file)
383    try:
384        with importer_lock:  # Acquires lock, raises LockError if fails
385            logger.info(f"{LOG_PREFIX}: Lock acquired. Proceeding with import.")
386
387            conn: Optional[sqlite3.Connection] = None
388            retries = 0
389            while retries <= MAX_RETRIES:
390                try:
391                    conn = get_db_connection(db_path)
392                    # Ensure table exists
393                    create_table_if_not_exists(
394                        conn, table_name, final_mapping_file_path
395                    )
396                    break  # Connection successful
397                except (
398                    DatabaseError
399                ) as e:  # Errors here will have LOG_PREFIX_DB from underlying functions
400                    logger.error(
401                        f"{LOG_PREFIX}: Database connection or initial table setup error: {e}"
402                    )  # General error for the step
403                    if retries == MAX_RETRIES:
404                        logger.critical(
405                            f"{LOG_PREFIX_DB}: Max retries reached for DB connection/setup. Aborting."
406                        )
407                        return False
408
409                    backoff_time = min(
410                        INITIAL_BACKOFF_SECONDS * (2**retries), MAX_BACKOFF_SECONDS
411                    )
412                    logger.info(
413                        f"{LOG_PREFIX_DB}: Retrying DB connection/setup in {backoff_time} seconds... (Attempt {retries + 1}/{MAX_RETRIES})"
414                    )
415                    time.sleep(backoff_time)
416                    retries += 1
417
418            if not conn:  # Should not happen if loop logic is correct
419                logger.critical(
420                    f"{LOG_PREFIX_DB}: Failed to establish database connection after retries."
421                )
422                return False
423
424            already_imported = get_already_imported_files(imported_log_path)
425
426            if not sql_files_dir.exists() or not any(sql_files_dir.iterdir()):
427                logger.info(
428                    f"{LOG_PREFIX}: SQL files directory {sql_files_dir} is empty or does not exist. No files to import."
429                )
430                if conn:
431                    conn.close()
432                return True
433
434            sql_files_to_import = sorted(
435                [
436                    f
437                    for f in sql_files_dir.glob("*.sql")
438                    if f.is_file() and f.name not in already_imported
439                ]
440            )
441
442            if not sql_files_to_import:
443                logger.info(
444                    f"{LOG_PREFIX}: No new SQL files found to import in {sql_files_dir}."
445                )
446                if conn:
447                    conn.close()
448                return True
449
450            logger.info(
451                f"{LOG_PREFIX}: Found {len(sql_files_to_import)} new SQL file(s) to import."
452            )
453            total_files_processed = 0
454            total_files_failed = 0
455
456            for sql_file in sql_files_to_import:
457                logger.info(f"{LOG_PREFIX}: Processing SQL file: {sql_file.name}")
458                try:
459                    with open(sql_file, "r", encoding="utf-8") as f_sql:
460                        sql_script = f_sql.read()
461
462                    # SQLite's executescript handles transactions within the script.
463                    # If the script has BEGIN/COMMIT, it will be atomic.
464                    # If not, executescript itself doesn't wrap in a transaction by default.
465                    # Our exporter creates files with BEGIN TRANSACTION; ... COMMIT;
466                    conn.executescript(
467                        sql_script
468                    )  # This will commit if successful, or rollback if error within script
469                    conn.commit()  # Ensure changes from executescript are committed if it doesn't do it itself for some reason (it should)
470
471                    mark_file_as_imported(imported_log_path, sql_file.name)
472                    logger.info(
473                        f"{LOG_PREFIX}: Successfully imported {sql_file.name}"
474                    )  # General import log
475                    total_files_processed += 1
476                except sqlite3.Error as e_db:
477                    logger.error(
478                        f"{LOG_PREFIX_DB}: SQLite error processing file {sql_file.name}: {e_db}. The transaction in the file should have rolled back.",
479                        exc_info=True,
480                    )
481                    # Automatic rollback on error is a feature of how SQLite handles executescript with transactional SQL files.
482                    # No explicit rollback call needed here if the .sql file itself is transactional.
483                    total_files_failed += 1
484                except (
485                    IOError
486                ) as e_io:  # This is an I/O error for reading the .sql file, not DB specific
487                    logger.error(
488                        f"{LOG_PREFIX}: IO error reading file {sql_file.name}: {e_io}",
489                        exc_info=True,
490                    )
491                    total_files_failed += 1
492                except Exception as e_gen:  # General error, could be anything
493                    logger.error(
494                        f"{LOG_PREFIX}: Unexpected error processing file {sql_file.name}: {e_gen}",
495                        exc_info=True,
496                    )
497                    total_files_failed += 1
498
499            if conn:
500                logger.debug(f"{LOG_PREFIX_DB}: Closing database connection.")
501                conn.close()
502
503            logger.info(
504                f"{LOG_PREFIX}: SQL import process finished. Processed: {total_files_processed} files. Failed: {total_files_failed} files."
505            )
506            return total_files_failed == 0
507
508    except LockError:
509        logger.warning(
510            f"{LOG_PREFIX}: Could not acquire import lock. Another import process may be running. Exiting."
511        )  # Lock related, uses general LOG_PREFIX
512        return False  # Indicate failure due to locking
513    except Exception as e:
514        logger.critical(
515            f"{LOG_PREFIX}: Unhandled exception during SQL import: {e}", exc_info=True
516        )  # General import error
517        return False
518    finally:
519        # Ensure lock is released if acquired and an unexpected error occurred before __exit__
520        if importer_lock._lock_file_handle:  # Check if lock was acquired
521            importer_lock.release()

Main function to perform the SQL import process.
